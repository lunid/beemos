DELIMITER $$


DROP TRIGGER /*!50032 IF EXISTS */ `update_assinatura`$$

CREATE
    /*!50017 DEFINER = 'root'@'localhost' */
    TRIGGER `update_cliente_to_user` AFTER UPDATE ON `SVIP_ASSINATURA` 
    FOR EACH ROW BEGIN
	SELECT COUNT(1) INTO @QTD FROM SVIP_ASSINATURA_CONFIG WHERE ID_ASSINATURA = OLD.ID_ASSINATURA;
    
	IF(@QTD = 0) THEN
		
		INSERT INTO 
			SVIP_ASSINATURA_CONFIG
			(
				ID_ASSINATURA,
				AMBIENTE,
				DATA_REGISTRO
			)
		VALUES
			(
				OLD.ID_ASSINATURA,
				'TEST',				
				NOW()
			);			
	END IF;
END;
$$

DELIMITER ;
